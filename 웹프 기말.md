정규식
[abc] abc 중 하나
[^abc] abc를 제외한 모든 문자
[a-zA-Z\d@#$] 모든 영문자와 숫자, 특수문자 @#$
(x|y) x 또는 y
n+ n이 1번 이상
n* n이 0번 이상
n? n이 0 또는 1번
n{x} n이 x번
n{x,y} n이 x번 이상 y번 이하
n{x,} n이 x번 이상
n$ n으로 끝
^n n으로 시작
(?=n) n이 뒤따르는 문자열
(?!n) n이 뒤따르지 않는 문자열

서버 구동

const express = require("express"); / express 모듈 불러움
const app = express(); / express 앱 객체 생성
const fs = require("fs"); 파일 읽기/쓰기를 위한 node.js 내장 모듈 불러옴
app.set("port", process.env.PORT || 8080); / 서버 포트 설정(있으면 사용, 없으면 8080)
app.set("host", "ip주소"); / 호스트 주소 설정
app.use(express.static(__dirname)); / 현재 폴더의 html css js 파일을 클라이언트에게 제공
app.use(express.json()); / json형식으로 보낸 요청(body)를 파싱
app.use(express.urlencoded({extended: false})); / html(query) 태그 데이터를 파싱
app.get("/", (req, res) => {
    res.redirect("playjs_p2.html");
}); / html페이지로 자동 이동
app.listen(app.get("port"), "0.0.0.0", () => {
    console.log("Express server running at" + app.get("port") + app.get("host"));
}); / 서버를 0.0.0.0(모든 네트워크 인터페이스)에서 8080포트로 실행, 콘솔 출력

app.get("/getNotes", (req, res) => { / 파일 조회
    fs.readFile("./data/note.json", "utf-8", (err, data) => { / 파일 읽어서 data 배열 저장
        if(err) {
            console.log(err);
            return res.status(500).json({
                message: "파일 읽기 실패"
            });
        } / 예외처리
        res.type("application/json"); / res(출력)하는 형식이 JSON 형식
        res.send(data); / 반환
    });
});
app.post("/saveNote", (req, res) => { / 데이터 전송/저장
    fs.readFile("./data/note.json", "utf-8", (err, data) => { / 파일 읽어서 data배열 저장
        if(err){
            console.log(err);
            return res.status(500).json({
                message: "파일 읽기 실패"
            });
        } / 예외처리
        const newNote = req.body; / post방식 -> body가 내용
        const notes = JSON.parse(data); / data를 json형식으로 바꿈
        notes.push(newNote); / 파일에 추가
        const noteList = JSON.stringify(notes); / data를 문자열 형식으로 바꿈
        fs.writeFile("./data/note.json", noteList, "utf-8", (err) => { / noteList를 파일에 덮어씌움, 문자열만 가능
            if (err) {
                console.log(err);
                return res.status(500).send("파일 저장 실패");
            } / 예외처리
            res.send(noteList); / 반환
        });
    });
});
